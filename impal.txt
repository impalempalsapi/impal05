CREATE TABLE CUSTOMER(
	idCust int(10) not null,
	username varchar(20),
	password varchar(20),
	NamaCust varchar(100),
	JenisKelamin varchar(10),
	Alamat string,
	constrain idCust AutoIncreament Primary key
);

CREATE TABLE SPAREPART(
id_sparepart int(10) not null,
nama_sparepart VARCHAR(30),
tipe_hp VARCHAR(30) ,
module_hp VARCHAR(30),
harga_sparepart double
constrain id_sparepart AutoIncreament Primary key
);

CREATE TABLE HANDPHONE ( 
tipe_hp varchar(10) NOT NULL,
merek_hp varchar(10) NOT NULL,
harga_hp NUMBER(10) NOT NULL,
deskripsi varchar(20) NOT NULL,
PRIMERY KEY (tipe_data)
);

CREATE TABLE REPARASI(
idCust int (10) Foreign key references Customer(idCust),
id_sparepart int (10) Foreign key references Sparepart(id_sparepart)
);

CREATE TABLE MEMBELI(
	idCust int(10) Foreign key references Customer(idCust),
	tipeHP varchar(10) foreign key references Handphone(tipeHP)
);
